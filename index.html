/* ═══════════════════════════════════════════════ MORNING DASHBOARD — Google Apps Script Handles both POST (write) and GET (read today) ═══════════════════════════════════════════════ */ /* ── GET: read today's data back to the dashboard ── */ function doGet(e) { const action = e.parameter.action; if (action === 'getToday') { const dateISO = e.parameter.date; if (!dateISO) return jsonOut({ status: 'error', message: 'No date provided' }); const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Morning Reflections ── const mrSheet = ss.getSheetByName('Morning Reflections'); let morningData = null; if (mrSheet && mrSheet.getLastRow() > 1) { const rows = mrSheet.getDataRange().getValues(); // Col index: 0=Date, 1=DateISO, 2=Time, 3=Priority, 4=Emotional Intention, 5=Gratitude, 6=Checklist Done, 7=Score for (let i = rows.length - 1; i >= 1; i--) { if (String(rows[i][1]).trim() === dateISO) { morningData = { date: rows[i][0], dateISO: rows[i][1], time: rows[i][2], priority: rows[i][3], emotional_intention: rows[i][4], gratitude: rows[i][5], checklist_done: rows[i][6], score: rows[i][7], }; break; // take the most recent entry for today } } } // ── Situations for today ── const sitSheet = ss.getSheetByName('Situations'); const todaySituations = []; if (sitSheet && sitSheet.getLastRow() > 1) { const rows = sitSheet.getDataRange().getValues(); // Col index: 0=Date, 1=DateISO, 2=Situation, 3=Trigger, 4=Reaction, 5=Emotion, 6=Strategy, 7=Logged At for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]).trim() === dateISO) { todaySituations.push({ date: rows[i][0], dateISO: rows[i][1], title: rows[i][2], trigger: rows[i][3], reaction: rows[i][4], emotion: rows[i][5], strategy: rows[i][6], loggedAt: rows[i][7], }); } } } return jsonOut({ status: 'ok', morning: morningData, situations: todaySituations, }); } // Default: health check return jsonOut({ status: 'alive', ts: new Date().toISOString() }); } /* ── POST: write data from the dashboard ── */ function doPost(e) { try { // Accepts URLSearchParams (e.parameter.payload) OR raw JSON body const raw = (e.parameter && e.parameter.payload) ? e.parameter.payload : e.postData.contents; const data = JSON.parse(raw); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetName = data.sheet || 'Morning Reflections'; if (sheetName === 'Situations') { writeSituation(ss, data); } else { writeMorningReflection(ss, data); } return jsonOut({ status: 'ok' }); } catch (err) { return jsonOut({ status: 'error', message: err.toString() }); } } /* ── Write morning reflection row ── */ function writeMorningReflection(ss, data) { let sheet = ss.getSheetByName('Morning Reflections'); if (!sheet) { sheet = ss.insertSheet('Morning Reflections'); sheet.appendRow([ 'Date', 'Date ISO', 'Time', 'Priority', 'Emotional Intention', 'Gratitude', 'Checklist Done', 'Score', 'Saved At' ]); sheet.setFrozenRows(1); styleHeader(sheet); } // Check if a row for this dateISO already exists — update it instead of appending const dateISO = data.dateISO || ''; if (dateISO) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]).trim() === dateISO) { // Update existing row in place sheet.getRange(i + 1, 1, 1, 9).setValues([[ data.date || rows[i][0], dateISO, data.time || rows[i][2], data.priority || rows[i][3], data.emotional_intention || rows[i][4], data.gratitude || rows[i][5], data.checklist_done || rows[i][6], data.score || rows[i][7], new Date().toLocaleString(), ]]); return; } } } // No existing row — append new sheet.appendRow([ data.date || '', dateISO, data.time || '', data.priority || '', data.emotional_intention || '', data.gratitude || '', data.checklist_done || '', data.score || '', new Date().toLocaleString(), ]); } /* ── Write situation row ── */ function writeSituation(ss, data) { let sheet = ss.getSheetByName('Situations'); if (!sheet) { sheet = ss.insertSheet('Situations'); sheet.appendRow([ 'Date', 'Date ISO', 'Situation', 'Trigger', 'Reaction', 'Emotion', 'Strategy', 'Logged At' ]); sheet.setFrozenRows(1); styleHeader(sheet); } // If it's a strategy update, find the matching row and update strategy column only if (data.action === 'update_strategy' && data.dateISO && data.title) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]).trim() === data.dateISO && String(rows[i][2]).trim() === String(data.title).trim()) { sheet.getRange(i + 1, 7).setValue(data.strategy || ''); // col G = Strategy return; } } } // New situation — append sheet.appendRow([ data.date || '', data.dateISO || '', data.title || '', data.trigger || '', data.reaction || '', data.emotion || '', data.strategy || '', new Date().toLocaleString(), ]); } /* ── Header styling ── */ function styleHeader(sheet) { const cols = sheet.getLastColumn(); const r = sheet.getRange(1, 1, 1, cols); r.setBackground('#1a1a2e'); r.setFontColor('#C9A84C'); r.setFontWeight('bold'); sheet.setColumnWidths(1, cols, 200); } /* ── JSON output helper ── */ function jsonOut(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
